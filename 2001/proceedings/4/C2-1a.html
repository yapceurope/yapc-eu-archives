    Data initialization section (using symbolic references)

        while (<>) {
        chomp;
        push @elp, $_;
        m/^99/ || next;

        # Convert the expanded LP-list to elements known in the format(s)
        foreach $lp (@elp) {
            ($cat, $type, $data) = split m/\|/, $lp, 3;
            $data =~ s/\s+$//;
            $ect  = "$cat$type";
            $e    = "e$ect";
            $$e   = $data;
            }

        # format this entity
        write;

        reset "e";
        }
